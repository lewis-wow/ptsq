# Introspection query

```ts title="server.ts"
import { createServer } from '@ptsq/server';

createServer({
  introspection: '*',
});
```

## Introspection query for only some hosts

```ts title="server.ts"
import { createServer } from '@ptsq/server';

createServer({
  introspection: ['http://localhost:3000' /* ... */],
});
```

## CLI

```bash
instrospection-cli --url="http://localhost:4000/ptsq/introspection" --out="schema.generated.ts"
```
